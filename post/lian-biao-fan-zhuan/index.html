链表反转

2021-06-03 算法 链表 递归

😭算法太难了 ，基本的一个做了好久 ，还抄了答案

// 链表结构
public class ListNode {
    int val;
    ListNode next;
    ListNode(int x){val =x;}
}

第一步 ：反转整个链表

/**
     * 递归反转整理链表
     * @param head
     * @return
     * 1->2->3
     * 1->2 第一步 返回一个 2
     以 1->2->null 为例子
     */
    ListNode reverse(ListNode head){
        if (head.next == null){return head;}
        ListNode listHead =reverse(head.next);// 返回2  2->null
        head.next.next =head;// 1->2 ->null   那么将head（1） 给 2->1
        head.next =null;
        return listHead;
    }

反转整个链表（2）

// 反转的 另一种方法
    public  ListNode reverseList(ListNode head){
        //  1- >  2  -> 3
        // 前一个节点
        ListNode pre =null;
        // 当前节点 1
        ListNode cur =head;
        // 下一个节点
        ListNode next=null;

        while (cur != null){
            // 下一个节点  =  当前节点的下一个节点
            // next = 节点 2
            // next =节点 3
            // next = null
            next =cur.next;
            // 当前节点的下一个节点 =  前一个节点
            //  节点 1 .next = null
            // 节点 2.next =节点 1
            // 节点3 .next = 节点2
            cur.next=pre;
            // 前一个节点  = 当前节点
            // pre = 节点 1
            // pre = 节点 2
            // pre =节点 3
            pre =cur;
            // 当前节点 = 下一个节点
            // cur = 节点 2
            // cur =节点 3
            // cur =null
            cur=next;
        }
        return pre;
    }

第二步：反转前N个节点

    /**
     * 递归反转整理链表
     * @param head
     * @return
     * 1->2->3 为例子
     * 反转前两个
     * 第一步 ：   将2 下一个取出来  nextNode =3
     *  将 1 下一个 节点 2  的下一个节点  3  变为 1
     *  2 和 1 节点反转
     *  随后将 1 的下一个节点 变为nextNode
     *  2->1->3
     *  在返回 第二个节点 也就是 2
     */
    ListNode nextNode =null;
    ListNode reverse(ListNode head,int n){
//      if (head.next == null){
//
//          return head;
//      }
        if (n == 1){
            nextNode=head.next;
            return head;
        }
        ListNode listHead =reverse(head.next,n-1);// 返回2  2->null
        head.next.next =head;// 1->2 ->null   那么将head（1） 给 2->1
        //head.next =null;
        head.next =nextNode;
        return listHead;
    }

第三步： 反转n-m

public ListNode betweenListNode(ListNode head,int m ,int n){
        // 如果为1 以后 就是反转  链表的  前 n个节点
        if (m == 1){
            return reverse(head,n);
        }
        // 采用 m 减少到 1  以后  n的长度正好是 n-m+1 的
        // 也就是需要 反转的数量 和位置

        // head.next 目的是为了指向 反转的节点  （head.next 传入参数 ：将下一个需要验证是否反转的节点传入）
        head.next = betweenListNode(head.next, m - 1, n - 1);
        return head;
    }

反转n-m（2）

public  ListNode reverseList(ListNode head,int m ,int n){
        // 在头节点之前定义一个新的节点，preHead
        ListNode preHead = new ListNode(0);
        preHead.next = head;
        // 定义一个cur节点，让其从preHead开始，先找到第m-1个节点，作为curPre
        ListNode cur = preHead;
        for(int i = 1; i < m; i++)
            cur = cur.next;
        ListNode curPre = cur; // cur 为 m-1;
        cur = cur.next; // 现在cur为第m个节点
        // 从第m个节点到第n个节点，我们需要将n-m+1个节点进行翻转，那就是将n-m个节点插在第m个节点之前，因此需要n-m次操作
        // 进入循环之前需要定义一个curNext
        ListNode curNext = null;
        for(int i = 0; i < n-m; i++){
            // 这个地方也可写成 for(int i = m; i < n; i++)
            curNext = cur.next;
            cur.next = curNext.next;
            curNext.next = curPre.next;
            curPre.next = curNext;
        }
        // 结束后，输出preHead.next即可
        return preHead.next;
    }
