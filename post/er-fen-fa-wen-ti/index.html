public int search(int[] nums, int target) {
   int end,start =0,middle=0;
   int length = nums.length-1;
   end =length;
   while (end >= start){
      middle =start + (end-start)/2;
      if (nums[middle] == target)return middle;
      if (nums[middle]  >target){
         end =middle -1;
      }else {
         start =middle +1;

}
